% --- plainbreak: line-break version of plain (body italic, head bold) ---
\newtheoremstyle{plainbreak}%
  {\topsep}%   space above the head
  {\topsep}%   space below the body
  {\itshape}%  body font
  {}%          indent
  {\bfseries}% head font
  {}%          punctuation after the head
  {\newline}%  line break after the head
  {\thmname{#1}\thmnumber{ #2}\thmnote{: #3}}%
% --- definitionbreak: line-break version of definition (body upright, head bold) ---
\newtheoremstyle{definitionbreak}%
  {\topsep}%
  {\topsep}%
  {\upshape}%  body is upright
  {}%
  {\bfseries}%
  {}%
  {\newline}%
  {\thmname{#1}\thmnumber{ #2}\thmnote{: #3}}%
% --- definitioninline: run-in version of definitionbreak (no line break after the head) ---
%     The head is only "name number" (no title, no punctuation); the body runs on
%     right after the head.  For exercises (exc): "Exercise N.M  <statement>".
\newtheoremstyle{definitioninline}%
  {\topsep}%
  {\topsep}%
  {\upshape}%  body is upright
  {}%
  {\bfseries}% head font
  {}%          no punctuation after the head
  { }%         space after the head (no line break)
  {\thmname{#1}\thmnumber{ #2}}% do not print the title (note)
% --- remarkbreak: line-break version of remark (body upright, head italic, reduced spacing) ---
\newtheoremstyle{remarkbreak}%
  {0.5\topsep}%  remark-type styles use half the vertical spacing
  {0.5\topsep}%
  {\upshape}%
  {}%
  {\itshape}%    head is italic (not bold)
  {}%
  {\newline}%
  {\thmname{#1}\thmnumber{ #2}\thmnote{: #3}}%
% --- lawbreak: for laws (title only as head, no number, line break after the head) ---
\newtheoremstyle{lawbreak}%
  {\topsep}%
  {\topsep}%
  {\upshape}%   body is upright
  {}%
  {\bfseries}%  head bold
  {}%           no punctuation (the title is the head itself)
  {\newline}%
  {\thmnote{#3}}%   <- display only the note as the head
% --- solution: for answers.  Same run-in as amsthm's standard definition, but
%     the space above the head is halved so it sits close to the preceding
%     exercise (box). ---
%     en keeps the `.` after the head ("Solution."); jp drops it, like exc.
\newtheoremstyle{solution}%
  {0.5\topsep}% space above the head (smaller than definition's \topsep)
  {\topsep}%    space below the body
  {\normalfont}% body in normal font
  {}%

  {.}%          head punctuation ("Solution.")
  {.5em}%       space after the head (run-in)
  {}%           default head format (name number)
% --- Make the line break after the head work even when the theorem body starts with a list ---
%     amsthm's \newline style stores the head in the box \@labels and attaches it
%     via \everypar (\dth@everypar) at the start of the next paragraph, but when
%     the body starts directly with a list environment, the kernel's
%     \@donoparitem joins the head box with the first \item on the same line,
%     and the break penalty (\nobreak\hfil\break) sealed inside the box has no
%     effect. The fix is two-fold:
%     (1) Do not put \thmheadnl and \hskip\thm@headsep inside the head box;
%         execute them outside the box at the start of the paragraph (redefine
%         \@begintheorem and \dth@everypar). For ordinary text bodies the
%         output is identical to before.
%     (2) Right before a list environment starts, if a head is pending, attach
%         it by forcing a paragraph with \leavevmode while \thmheadnl is
%         temporarily disabled (no break node is emitted in the first place;
%         LuaTeX-ja inserts whatsits around boxes and \hskip, so removing the
%         nodes afterwards with \unskip\unpenalty is impossible). The list then
%         closes the paragraph, so the head forms its own line and the list
%         starts below it.
\makeatletter
\newif\ifcallout@thmhead
\let\callout@orig@deferredthmhead\deferred@thm@head
\def\deferred@thm@head#1{%
  \global\callout@thmheadtrue
  \callout@orig@deferredthmhead{#1}%
}
% move \thmheadnl \hskip\thm@headsep out of the head box of amsthm's \@begintheorem
\def\@begintheorem#1#2[#3]{%
  \deferred@thm@head{\the\thm@headfont \thm@indent
    \@ifempty{#1}{\let\thmname\@gobble}{\let\thmname\@iden}%
    \@ifempty{#2}{\let\thmnumber\@gobble}{\let\thmnumber\@iden}%
    \@ifempty{#3}{\let\thmnote\@gobble}{\let\thmnote\@iden}%
    \thm@swap\swappedhead\thmhead{#1}{#2}{#3}%
    \the\thm@headpunct
  }%
  \ignorespaces}
% add the relocated commands and the flag reset to amsthm's \dth@everypar
\dth@everypar={%
  \@minipagefalse \global\@newlistfalse
  \@noparitemfalse
  \if@inlabel
    \global\@inlabelfalse
    \begingroup \setbox\z@\lastbox
     \ifvoid\z@ \kern-\itemindent \fi
    \endgroup
    \unhbox\@labels
    \thmheadnl\hskip\thm@headsep % <- executed outside the box (relocated)
  \fi
  \if@nobreak \@nobreakfalse \clubpenalty\@M
  \else \clubpenalty\@clubpenalty \everypar{}%
  \fi
  \global\callout@thmheadfalse % <- clear the flag once the head is attached
}%
\def\callout@nlmarker{\newline}
\newcommand*{\callout@fixthmheadbreak}{%
  \ifcallout@thmhead
    \ifx\thmheadnl\callout@nlmarker % only heads of the line-break styles
      \begingroup
        \let\thmheadnl\relax
        \thm@headsep\z@skip
        \leavevmode % attach the head without a break (the list closes the paragraph)
      \endgroup
    \fi
  \fi
}

% --- Tighten list spacing inside boxes ---
%     The class defaults (ltjsarticle: \leftmargini=3\zw, \topsep=0.5\baselineskip;
%     scrartcl etc.: \leftmargini=2.5em with large topsep/itemsep) are excessive
%     inside boxes with narrow padding, so override the first level only, locally
\makeatletter
\newcommand*{\callout@tightlists}{%
  \ifdefined\zw \leftmargini=1.5\zw \else \leftmargini=1.5em \fi % \zw is a LuaTeX-ja-only unit
  \def\@listi{\leftmargin\leftmargini
    \labelwidth\leftmargini \advance\labelwidth-\labelsep
    \parsep\z@
    \topsep 0.2\baselineskip
    \itemsep\z@\relax}%
}

% does the name contain * ?  (yes -> #2, no -> #3)
\def\callout@ifstarred#1{\callout@ifstarred@#1*\@nil}
\def\callout@ifstarred@#1*#2\@nil{\ifblank{#2}{\@secondoftwo}{\@firstoftwo}}
% \RegisterCalloutEnv{env name}{frame color}[title mode: optional / mandatory]
%   Registers the environment for use with callout.  The unnumbered variant
%   "name*" is registered automatically with the same settings (nothing is
%   done if the name itself ends with *).  Explicit registrations are
%   recorded and always take precedence over automatic ones, regardless of
%   order.  Registration only adds to the lookup table; the environment
%   itself must be defined separately (use \NewCalloutThmEnv /
%   \NewCalloutLawEnv below to define and register in one go).
\NewDocumentCommand{\RegisterCalloutEnv}{ m m O{optional} }{%
  \callout@register{#1}{#2}{#3}%
  \@namedef{callout@explicit@#1}{}%
  \callout@ifstarred{#1}{}{%
    \ifcsname callout@explicit@#1*\endcsname\else
      \callout@register{#1*}{#2}{#3}%
    \fi
  }%
}
% --- helpers to create theorem-type environments: define + register ---
% \NewCalloutThmEnv{env name}[shared counter]{display name}{frame color}[reset unit]
%   Creates a numbered environment and an unnumbered one ("env name*"), and
%   registers both.  The options follow the same (mutually exclusive) rules
%   as \newtheorem:
%     [shared counter]  ... share numbering with the given environment
%                           (e.g. [thm])
%     [reset unit]      ... independent numbering, reset per counter
%                           (e.g. [section] / [subsection])
%     both omitted      ... independent numbering, never reset (1, 2, ...)
%   Like \newtheorem, the current \theoremstyle is used.
\NewDocumentCommand{\NewCalloutThmEnv}{ m o m m o }{%
  \IfValueTF{#2}
    {\IfValueTF{#5}
       {\GenericError{}{\string\NewCalloutThmEnv: shared counter and reset
          unit cannot be specified together}{}{}}
       {\newtheorem{#1}[#2]{#3}}}
    {\IfValueTF{#5}
       {\newtheorem{#1}{#3}[#5]}
       {\newtheorem{#1}{#3}}}%
  \newtheorem*{#1*}{#3}%
  \RegisterCalloutEnv{#1}{#4}% the * variant is registered automatically
}
% \NewCalloutLawEnv{env name}{frame color}
%   Creates and registers a law-type environment (mandatory title, no number,
%   lawbreak style showing only the title as the heading).  No * variant.
\NewDocumentCommand{\NewCalloutLawEnv}{ m m }{%
  \edef\callout@savedthmstyle{\the\thm@style}% \thm@style is a token register
  \theoremstyle{lawbreak}%
  \newtheorem*{#1@inner}{#1}% internal; the name is never shown (\thmname unused)
  \expandafter\thm@style\expandafter{\callout@savedthmstyle}% restore style
  \NewDocumentEnvironment{#1}{ m }{%
    \begin{#1@inner}[##1]%
  }{%
    \end{#1@inner}%
  }%
  \callout@register{#1}{#2}{mandatory}% low-level: no * variant to register
  \@namedef{callout@explicit@#1}{}%
}

% --- callout: for wrapping theorem-like environments ---
%     the optional title is passed as the optional argument (theorem note) of the inner environment
\makeatletter
\NewDocumentEnvironment{callout}{ o m }{%
  \ifcsname callout@color@#2\endcsname\else
    \GenericError{}{callout environment: `#2' is not registered.
      Register it with \RegisterCalloutEnv}{}{}%
  \fi
  \begin{calloutbox}[\@nameuse{callout@color@#2}]%
  \ifcsstring{callout@titlemode@#2}{mandatory}
    {% environment that requires a title
     \IfValueTF{#1}
       {\begin{#2}{#1}}
       {\GenericError{}{callout environment: `#2' requires a title}{}{}%
        \begin{#2}{??}}%
    }
    {% environment with an optional title
     \IfValueTF{#1}
       {\begin{#2}[#1]}
       {\begin{#2}}%
    }%
}{%
  \end{#2}%
  \end{calloutbox}%
}
\makeatother

\NewDocumentCommand{\NewCalloutBoxEnv}{ m m o }{%
  % #1: environment name, #2: color, #3: default title (optional)
  \IfValueTF{#3}
    {% with a default title: use #3 when the environment's title is omitted
     \NewDocumentEnvironment{#1}{ O{#3} }{%
       \begin{calloutbox}[#2][##1]%
     }{%
       \end{calloutbox}%
     }%
    }
    {% without a default title: a box with no title when omitted
     \NewDocumentEnvironment{#1}{ o }{%
       \IfValueTF{##1}
         {\begin{calloutbox}[#2][##1]}
         {\begin{calloutbox}[#2]}%
     }{%
       \end{calloutbox}%
     }%
    }%
}
